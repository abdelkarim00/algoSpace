Sieve of Eratosthenes Algorithm

üìò Overview

The Sieve of Eratosthenes is an ancient algorithm used to find all prime numbers up to a specified integer. It works by iteratively marking the multiples of each prime number starting from 2.

üîç How It Works

  1. Create a list of consecutive integers from 2 to n.
  2. Initially, assume all numbers are prime.
  3. Starting from 2, mark all multiples of 2 as non-prime.
  4. Move to the next unmarked number and mark all its multiples as non-prime.
  5. Repeat until you've processed numbers up to ‚àön.
  6. The remaining unmarked numbers are prime.

‚è±Ô∏è Time Complexity

  • Time Complexity: O(n log log n) ‚Äî The algorithm efficiently marks non-prime numbers, with the marking process for each prime number being proportional to n/log n.

üíæ Space Complexity

  • Space Complexity: O(n) ‚Äî Requires an array of size n to store the primality of each number.

‚úÖ Easy Problems Solvable by Sieve of Eratosthenes

1. Finding All Prime Numbers Up to n

Given an integer n, print all prime numbers less than or equal to n.


#include <iostream>
#include <vector>

void sieve(int n) {
    std::vector<bool> prime(n+1, true);
    prime[0] = prime[1] = false;
    for (int p = 2; p * p <= n; ++p) {
        if (prime[p]) {
            for (int i = p * p; i <= n; i += p) {
                prime[i] = false;
            }
        }
    }
    for (int p = 2; p <= n; ++p) {
        if (prime[p]) {
            std::cout << p << " ";
        }
    }
    std::cout << std::endl;
}

int main() {
    int n = 30;
    std::cout << "Primes up to " << n << ": ";
    sieve(n);
    return 0;
}

2. Checking if a Number is Prime

Given an integer n, determine if it is a prime number.


#include <iostream>
#include <vector>

bool isPrime(int n) {
    if (n < 2) return false;
    std::vector<bool> prime(n+1, true);
    prime[0] = prime[1] = false;
    for (int p = 2; p * p <= n; ++p) {
        if (prime[p]) {
            for (int i = p * p; i <= n; i += p) {
                prime[i] = false;
            }
        }
    }
    return prime[n];
}

int main() {
    int n = 29;
    if (isPrime(n)) {
        std::cout << n << " is prime." << std::endl;
    } else {
        std::cout << n << " is not prime." << std::endl;
    }
    return 0;
}

1. Finding All Prime Numbers in a Range [m, n]

Given two integers m and n, print all prime numbers in the range [m, n].


#include <iostream>
#include <vector>

void sieve(int n, std::vector<bool>& prime) {
    prime.assign(n+1, true);
    prime[0] = prime[1] = false;
    for (int p = 2; p * p <= n; ++p) {
        if (prime[p]) {
            for (int i = p * p; i <= n; i += p) {
                prime[i] = false;
            }
        }
    }
}

void printPrimesInRange(int m, int n) {
    std::vector<bool> prime;
    sieve(n, prime);
    for (int i = m; i <= n; ++i) {
        if (prime[i]) {
            std::cout << i << " ";
        }
    }
    std::cout << std::endl;
}

int main() {
    int m = 10, n = 50;
    std::cout << "Primes in range [" << m << ", " << n << "]: ";
    printPrimesInRange(m, n);
    return 0;
}

2. Counting the Number of Primes Less Than or Equal to n

Given an integer n, count the number of prime numbers less than or equal to n.


#include <iostream>
#include <vector>

int countPrimes(int n) {
    std::vector<bool> prime(n+1, true);
    prime[0] = prime[1] = false;
    for (int p = 2; p * p <= n; ++p) {
        if (prime[p]) {
            for (int i = p * p; i <= n; i += p) {
                prime[i] = false;
            }
        }
    }
    int count = 0;
    for (int p = 2; p <= n; ++p) {
        if (prime[p]) {
            ++count;
        }
    }
    return count;
}

int main() {
    int n = 30;
    std::cout << "Number of primes less than or equal to " << n << ": " << countPrimes(n) << std::endl;
    return 0;
}

üß† Advanced Problem Solvable by Sieve of Eratosthenes

1. Finding the Prime Factorization of a Number

Given an integer n, find its prime factorization.


#include <iostream>
#include <vector>

void sieve(int n, std::vector<int>& spf) {
    spf.assign(n+1, 0);
    for (int i = 2; i <= n; ++i) {
        if (spf[i] == 0) {
            spf[i] = i;
            for (int j = i * i; j <= n; j += i) {
                if (spf[j] == 0) {
                    spf[j] = i;
                }
            }
        }
    }
}

void primeFactorization(int n, const std::vector<int>& spf) {
    while (n != 1) {
        std::cout << spf[n] << " ";
        n /= spf[n];
    }
    std::cout << std::endl;
}

int main() {
    int n = 60;
    std::vector<int> spf;
    sieve(n, spf);
    std::cout << "Prime factorization of " << n << ": ";
    primeFactorization(n, spf);
    return 0;
}
